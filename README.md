// 입력을 만드는 방법 : (매개) 변수를 선언

}

int main()
{
	int a1 = 10;

	int a1 = 10;

	std::cout << "res : " << MyFunc(a1, b1) << std::endl;

int MyFunc(int a, int b)   // 입력도있고 출력도 있는 함수 (함수 정의)
